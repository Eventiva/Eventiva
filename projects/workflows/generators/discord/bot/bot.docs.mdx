## Use in your dev environment

```ts
import { NodeEnv } from '@bitdev/node.node-env';
import { NodeEnvTemplate, EntityTemplate, NodeModuleTemplate, ExpressAppTemplate } from '@bitdev/node/generator/bot';

class MyEnv extends NodeEnv {
  generators() {
    return TemplateList.from([
      NodeEnvTemplate.from(),
      EntityTemplate.from(),
      NodeModuleTemplate.from(),
      ExpressAppTemplate.from(),
    ]);
  }
}
```
