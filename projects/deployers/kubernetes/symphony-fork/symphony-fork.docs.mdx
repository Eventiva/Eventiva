A plugable SymphonyFork aspect. Used to maintain SymphonyFork across the platform.

## Get started

Compose the aspect into your Harmony platform:

```ts
import { SymphonyPlatformAspect } from '@bitdev/symphony.symphony-platform';

/**
 * compose the platform.
 */
export const MyPlatform = HarmonyPlatform.from({
  name: 'my-platform',
  platform: SymphonyPlatformAspect,
  runtimes: [
    new BrowserRuntime(),
    new NodeJSRuntime()
   ],

  aspects: [[
    SymphonyForkAspect, {
      auth: {
        basic: {
          server: 'https://1.2.3.4',
          certificate: process.env.KUBE_CERTIFICATE_AUTHORITY_DATA,
          token: process.env.KUBE_USER_TOKEN
        }
      },
      docker: {
        imagePrefix: 'dockerhub-account'
        username: 'ci-user',
        password: process.env.DOCKERHUB_PASSWORD,
      }
  }]],
});

export default MyPlatform;
```
